4-2) 내가 기여한 내용
```python
import matplotlib.pyplot as plt
import numpy as np
import math as m

a = input('함수(f)를 입력하세요')
b = input('함수(g)를 입력하세요')

x_min = input('하한')
x_max = input('상한')

dot_x = []
dot_y = []

ar_xn = range(int(x_min), int(x_max)+1)
ar_x = np.arange(int(x_min), int(x_max),0.1)
ar_x[-1] = int(x_max)

ar1_y = []
ar2_y = []

for i in ar_xn :
    x = i
    f = eval(a)
    g = eval(b)
    f = float(f)
    g = float(g)
    if f-g >= 0 :
        for j in range(m.ceil(g), m.floor(f)+1) :
            dot_x.append(i)
            dot_y.append(j)
    else:    
        for j in range(m.ceil(f), m.floor(g)+1) :
            dot_x.append(i)
            dot_y.append(j)

for i in ar_x :
    x = i
    f = eval(a)
    g = eval(b)
    f = float(f)
    g = float(g)
    ar1_y.append(f)
    ar2_y.append(g)

cnt = 0
a = len(dot_x)
for i in range(a) :
    plt.scatter(dot_x[i],dot_y[i], color='r')
    cnt = cnt+1
plt.plot(ar_x,ar1_y)
plt.plot(ar_x,ar2_y)
plt.show()
print(cnt)
```
